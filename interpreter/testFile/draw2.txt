--------------- 函数f(t)=t的图形
origin is (20, 600);	-- 设置原点的偏移量
rot is 0;			-- 设置旋转角度(不旋转)
scale is (10, 10);		-- 设置横坐标和纵坐标的比例
for T from 0 to 200 step 0.01 draw (t, 0);
				-- 横坐标的轨迹（纵坐标为0）
for T from 0 to 150 step 0.01 draw (0, -t);
				-- 纵坐标的轨迹（横坐标为0）
for T from 0 to 120 step 0.01 draw (t, -t);
				-- 函数f(t)=t的轨迹
